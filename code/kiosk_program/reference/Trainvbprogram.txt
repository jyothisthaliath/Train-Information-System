        'call the function
        Call ProcessInput
     ElseIf Connected = True Then
        RecArray(UBound(RecArray)) = RecArray(UBound(RecArray)) & InBuff
    End If

        If InBuff = "+" Then
            RecArray(LBound(RecArray)) = "+"

End Sub
